//数据结构实验报告一 顺序表排序  降序
#include<stdio.h>
#include<stdlib.h>
#include<time.h>
typedef struct node{
	int *data;
	int length;
	int size;
}Link;
int main()
{
	int L,a,i,j;
	Link A;
	srand((unsigned)time(NULL));
	scanf("%d",&L);
	A.data=(int *)malloc(sizeof(int)*L);
	A.length=0;
	A.size=L;
	for(i=0;i<L;i++)
	{
		a=rand()%100+1;
		printf("%d ",a);
		A.data[i]=a;
		A.length++;
	}
	printf("\n");
	for(i=0;i<L-1;i++)
	{
		for(j=0;j<L-i-1;j++)
		{
			if(A.data[j]<A.data[j+1])
			{
				A.data[j]=A.data[j]^A.data[j+1];
				A.data[j+1]=A.data[j]^A.data[j+1];
				A.data[j]=A.data[j]^A.data[j+1];
			}
		}
	}
	for(i=0;i<L;i++)
		printf("%d\n",A.data[i]);
	return 0;
}

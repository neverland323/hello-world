#include<stdio.h>
#include<stdlib.h>
const int Max=30;
typedef struct node{
	char *str;
	int top;
	int n;
}Stack;
void Judge1(char *a,int n,Stack *S);
void Judge2(char *a,int n,Stack *S);
int main()
{
	Stack S;
	char a[Max];
	int i=0;
	scanf("%d",&S.n);
	S.str=(char *)malloc(sizeof(char)*S.n);
	S.top=-1;
	getchar();
	gets(a);
	if(S.n%2!=0)
		Judge1(a,S.n,&S);
	else
		Judge2(a,S.n,&S);
	return 0;
}
void Judge1(char *a,int n,Stack *S)
{
	char ch;
	int i=0;
	int flag=-1;
	ch=a[i];
	while(ch!='\0')
	{
		if(i<(n-1)/2)
		{
			S->top++;
			S->str[S->top]=ch;
		}
		else if(i==(n-1)/2)
		{}
		else
		{
			if(S->str[S->top]==ch)
			{
				flag=1;
				S->top--;
			}
			else
			{
				printf("F\n");
				break;
			}
		}
		i++;
		ch=a[i];
	}
	if(flag==1)
		printf("T\n");
}
void Judge2(char *a,int n,Stack *S)
{
	char ch;
	int i=0;
	int flag=-1;
	ch=a[i];
	while(ch!='\0')
	{
		if(i<=n/2-1)
		{
			S->top++;
			S->str[S->top]=ch;
		}
		else
		{
			if(S->str[S->top]==ch)
			{
				flag=1;
				S->top--;
			}
			else
			{
				printf("F\n");
				break;
			}
		}
		i++;
		ch=a[i];
	}
	if(flag==1)
		printf("T\n");
}

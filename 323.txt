//Printer Queue
#include<stdio.h>
const int MAX=128;
int n,m;
int qu[MAX],s;
int ans;
int main()
{
	int t;
	int i;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		m--;
		for(i=0;i<n;i++)
		{
			scanf("%d",&qu[i]);
		}
		ans=0;
		s=0;
		while(qu[m]>0)
		{
			if(qu[s]>0)
			{
				int max=0;
				for(i=0;i<n;i++)
				{
					if(max<qu[i])
						max=qu[i];
				}
				if(qu[s]>=max)
				{
					ans++;
					qu[s]=0;
				}
			}
			s=(s+1)%n;
		}
		printf("%d\n",ans);
	}
	return 0;
}
